function renameKey(obj, old_key, new_key) {   
    // check if old key = new key  
    if (old_key !== new_key) {                  
      Object.defineProperty(obj, new_key, // modify old key
      // fetch description from object
      Object.getOwnPropertyDescriptor(obj, old_key));
      delete obj[old_key]; // delete old key
  }
}

var s = {
  "skilltree": []
}
var skilltree=document.getElementById("firstHeading").textContent.replace(/\s/g, '') //skilltree
renameKey(s,"skilltree",skilltree)

var skillbox=document.getElementsByClassName("skillbox") 
var name
var description
var cost
var wikiLink
var BP
for (let i = 0; i < skillbox.length; i++) {
  name=skillbox[i].children[0].children[0].innerHTML
  wikiLink=skillbox[i].children[0].children[0].href
  description=skillbox[i].getElementsByTagName("i")[0].textContent.replace(".",". ")
  cost=document.getElementsByClassName("skillDesc")[i].children[0].textContent.split("|")[0]
  cost = cost.replace(/\s/g, '')
  if (cost.includes(",")) {
    cost=cost.split(",")
    BP = cost[0]
    cost = cost[1]
  } else {
    BP = 0
  }
  s[skilltree].push({"name":name,"description":description,"cost":cost,"BP":BP,"wikiLink":wikiLink,"tier":"TIERHERE"})
} 
result = JSON.stringify(s,null, "\t")
result=result
